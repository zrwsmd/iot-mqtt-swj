## 4. 云服务器部署

### 4.1 准备云服务器

**确认 Java 环境**：

```bash
# SSH 登录云服务器
ssh root@your-server-ip

# 检查 Java 版本
java -version
# 输出：openjdk version "21.x.x"

# 如果未安装 Java 21
sudo apt-get update
sudo apt-get install -y openjdk-21-jdk
```

### 4.2 创建部署目录

```bash
# 创建目录
sudo mkdir -p /opt/iot-device
sudo mkdir -p /var/log/iot-device

# 设置权限
sudo chmod 755 /opt/iot-device
```

### 4.3 上传文件到云服务器

**方式一：使用 scp（推荐）**

```bash
# 在本地电脑执行（PowerShell 或 CMD）
cd e:\wulianwnag\parkingLock\parkingLock1

# 上传 JAR 包
scp target\parkingLock1-1.0.0-jar-with-dependencies.jar root@your-server-ip:/opt/iot-device/

# 上传配置文件
scp device_id.json root@your-server-ip:/opt/iot-device/
```

**方式二：使用 FTP 工具**

使用 FileZilla、WinSCP 等工具上传：
- `parkingLock1-1.0.0-jar-with-dependencies.jar` → `/opt/iot-device/`
- `device_id.json` → `/opt/iot-device/`

### 4.4 云服务器运行 JAR

```bash
# SSH 登录云服务器
ssh root@your-server-ip

# 进入目录
cd /opt/iot-device

# 测试运行
java -jar parkingLock1-1.0.0-jar-with-dependencies.jar

# 预期输出：
# [DEBUG] HelloWorld - Hello world!
# [INFO] LinkKit - MQTT connecting...
# [INFO] LinkKit - MQTT connected
# [DEBUG] ThingSample - 注册服务处理器: set
# [DEBUG] ThingSample - 注册服务处理器: get
# [DEBUG] ThingSample - 注册服务处理器: restart
```

**如果连接成功，设备会在 IoT 控制台显示"在线"。**

### 4.5 后台运行 JAR

```bash
# 使用 nohup 后台运行
nohup java -jar parkingLock1-1.0.0-jar-with-dependencies.jar > /var/log/iot-device/output.log 2>&1 &

# 查看进程
ps aux | grep parkingLock

# 查看日志
tail -f /var/log/iot-device/output.log

# 停止进程
# 先找到进程 ID
ps aux | grep parkingLock
# 然后 kill
kill -9 <PID>
```

### 4.6 配置 systemd 服务（推荐）

**创建服务文件**：

```bash
sudo nano /etc/systemd/system/iot-device.service
```

**内容**：

```ini
[Unit]
Description=Aliyun IoT Device Service
After=network.target

[Service]
Type=simple
User=root
WorkingDirectory=/opt/iot-device
ExecStart=/usr/bin/java -jar /opt/iot-device/parkingLock1-1.0.0-jar-with-dependencies.jar
Restart=always
RestartSec=10
StandardOutput=append:/var/log/iot-device/output.log
StandardError=append:/var/log/iot-device/error.log

[Install]
WantedBy=multi-user.target
```

**启动服务**：

```bash
# 重载 systemd
sudo systemctl daemon-reload

# 启动服务
sudo systemctl start iot-device

# 查看状态
sudo systemctl status iot-device

# 开机自启
sudo systemctl enable iot-device

# 查看日志
sudo journalctl -u iot-device -f

# 停止服务
sudo systemctl stop iot-device

# 重启服务
sudo systemctl restart iot-device
```

### 5.5 HTTP 服务器代码（server.js）

```javascript
require('dotenv').config();
const express = require('express');
const cors = require('cors');
const Iot20180120 = require('@alicloud/iot20180120').default;
const OpenApi = require('@alicloud/openapi-client');

const app = express();
const PORT = process.env.PORT || 3000;

// 中间件
app.use(cors());
app.use(express.json());

// 初始化阿里云 IoT 客户端
const config = new OpenApi.Config({
    accessKeyId: process.env.ACCESS_KEY_ID,
    accessKeySecret: process.env.ACCESS_KEY_SECRET,
    endpoint: `iot.${process.env.IOT_REGION}.aliyuncs.com`
});
const client = new Iot20180120(config);

// ===== 1. 调用设备服务 =====
app.post('/api/device/service/:serviceId', async (req, res) => {
    try {
        const { serviceId } = req.params;
        const { args = {} } = req.body;

        console.log(`[API] 调用服务: ${serviceId}, 参数:`, args);

        const request = new Iot20180120.InvokeThingServiceRequest({
            iotInstanceId: process.env.IOT_INSTANCE_ID,
            productKey: process.env.PRODUCT_KEY,
            deviceName: process.env.DEVICE_NAME,
            identifier: serviceId,
            args: JSON.stringify(args)
        });

        const response = await client.invokeThingService(request);

        res.json({
            success: response.body.success,
            message: `服务 ${serviceId} 调用成功`,
            data: {
                messageId: response.body.data?.messageId,
                result: response.body.data
            }
        });
    } catch (error) {
        console.error('[API] 调用服务失败:', error.message);
        res.status(500).json({
            success: false,
            message: '调用服务失败',
            error: error.message
        });
    }
});

// ===== 2. 设置设备属性 =====
app.post('/api/device/property', async (req, res) => {
    try {
        const { properties } = req.body;

        if (!properties || typeof properties !== 'object') {
            return res.status(400).json({
                success: false,
                message: '属性参数格式错误'
            });
        }

        console.log('[API] 设置属性:', properties);

        const request = new Iot20180120.SetDevicePropertyRequest({
            iotInstanceId: process.env.IOT_INSTANCE_ID,
            productKey: process.env.PRODUCT_KEY,
            deviceName: process.env.DEVICE_NAME,
            items: JSON.stringify(properties)
        });

        const response = await client.setDeviceProperty(request);

        res.json({
            success: response.body.success,
            message: '属性设置成功',
            data: {
                messageId: response.body.data?.messageId
            }
        });
    } catch (error) {
        console.error('[API] 设置属性失败:', error.message);
        res.status(500).json({
            success: false,
            message: '设置属性失败',
            error: error.message
        });
    }
});

// ===== 3. 查询设备属性 =====
app.get('/api/device/property', async (req, res) => {
    try {
        console.log('[API] 查询设备属性');

        const request = new Iot20180120.QueryDevicePropertyStatusRequest({
            iotInstanceId: process.env.IOT_INSTANCE_ID,
            productKey: process.env.PRODUCT_KEY,
            deviceName: process.env.DEVICE_NAME
        });

        const response = await client.queryDevicePropertyStatus(request);
        const properties = response.body.data?.list?.propertyStatusInfo || [];

        const propertyMap = {};
        properties.forEach(prop => {
            propertyMap[prop.identifier] = {
                value: prop.value,
                time: new Date(prop.time).toLocaleString('zh-CN')
            };
        });

        res.json({
            success: true,
            message: '查询属性成功',
            data: {
                properties: propertyMap,
                count: properties.length
            }
        });
    } catch (error) {
        console.error('[API] 查询属性失败:', error.message);
        res.status(500).json({
            success: false,
            message: '查询属性失败',
            error: error.message
        });
    }
});

// ===== 4. 查询设备详情 =====
app.get('/api/device/detail', async (req, res) => {
    try {
        console.log('[API] 查询设备详情');

        const request = new Iot20180120.QueryDeviceDetailRequest({
            iotInstanceId: process.env.IOT_INSTANCE_ID,
            productKey: process.env.PRODUCT_KEY,
            deviceName: process.env.DEVICE_NAME
        });

        const response = await client.queryDeviceDetail(request);
        const deviceInfo = response.body.data;

        res.json({
            success: true,
            message: '查询设备详情成功',
            data: {
                deviceName: deviceInfo.deviceName,
                productKey: deviceInfo.productKey,
                status: deviceInfo.status,
                online: deviceInfo.online,
                gmtCreate: new Date(deviceInfo.gmtCreate).toLocaleString('zh-CN'),
                gmtActive: new Date(deviceInfo.gmtActive).toLocaleString('zh-CN'),
                gmtOnline: deviceInfo.gmtOnline ? new Date(deviceInfo.gmtOnline).toLocaleString('zh-CN') : 'N/A',
                ipAddress: deviceInfo.ipAddress || 'N/A'
            }
        });
    } catch (error) {
        console.error('[API] 查询设备详情失败:', error.message);
        res.status(500).json({
            success: false,
            message: '查询设备详情失败',
            error: error.message
        });
    }
});

// ===== 5. 获取设备状态（综合信息） =====
app.get('/api/device/status', async (req, res) => {
    try {
        console.log('[API] 获取设备状态');

        // 查询设备详情
        const detailRequest = new Iot20180120.QueryDeviceDetailRequest({
            iotInstanceId: process.env.IOT_INSTANCE_ID,
            productKey: process.env.PRODUCT_KEY,
            deviceName: process.env.DEVICE_NAME
        });
        const detailResponse = await client.queryDeviceDetail(detailRequest);
        const deviceInfo = detailResponse.body.data;

        // 查询设备属性
        const propertyRequest = new Iot20180120.QueryDevicePropertyStatusRequest({
            iotInstanceId: process.env.IOT_INSTANCE_ID,
            productKey: process.env.PRODUCT_KEY,
            deviceName: process.env.DEVICE_NAME
        });
        const propertyResponse = await client.queryDevicePropertyStatus(propertyRequest);
        const properties = propertyResponse.body.data?.list?.propertyStatusInfo || [];

        const propertyMap = {};
        properties.forEach(prop => {
            propertyMap[prop.identifier] = {
                value: prop.value,
                time: new Date(prop.time).toLocaleString('zh-CN')
            };
        });

        res.json({
            success: true,
            message: '获取设备状态成功',
            data: {
                online: deviceInfo.online,
                status: deviceInfo.status,
                properties: propertyMap,
                lastOnlineTime: deviceInfo.gmtOnline ? new Date(deviceInfo.gmtOnline).toLocaleString('zh-CN') : 'N/A'
            }
        });
    } catch (error) {
        console.error('[API] 获取设备状态失败:', error.message);
        res.status(500).json({
            success: false,
            message: '获取设备状态失败',
            error: error.message
        });
    }
});

// 健康检查
app.get('/health', (req, res) => {
    res.json({
        status: 'ok',
        timestamp: new Date().toISOString()
    });
});

// 根路径
app.get('/', (req, res) => {
    res.json({
        name: 'IoT Controller API',
        version: '1.0.0',
        endpoints: {
            health: 'GET /health',
            invokeService: 'POST /api/device/service/:serviceId',
            setProperty: 'POST /api/device/property',
            queryProperty: 'GET /api/device/property',
            queryDetail: 'GET /api/device/detail',
            getStatus: 'GET /api/device/status'
        }
    });
});

// 启动服务器
app.listen(PORT, () => {
    console.log(`\n🚀 IoT Controller API 运行中`);
    console.log(`📖 地址: http://localhost:${PORT}`);
    console.log(`💚 健康检查: http://localhost:${PORT}/health\n`);
});
```

require('dotenv').config();
const axios = require('axios');

const BASE_URL = 'http://localhost:3000';

async function testAllAPIs() {
    console.log('\n========== IoT API 测试 ==========\n');

    try {

        console.log('');

        // 2. 查询设备属性

        // 3. 设置设备属性

            properties: {
                ADASSwitch: 1
            }
        });
        console.log('   ✓ 设置结果:', setRes.data.success);
        console.log('   ✓ 消息ID:', setRes.data.data.messageId);
        console.log('');

        // 等待 2 秒让设备处理

        // 4. 调用设备服务

testAllAPIs();

# IoT 控制台 → 产品 → 功能定义 → 属性
